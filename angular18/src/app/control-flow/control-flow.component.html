Control Flow in Angular 18

IF Block

@if (a > b) {

a is greater than b

} @else if (b > a) {

a is less than b

} @else {

a is equal to b

}

FOR Block

@for (item of items; track item) {

Item: {{ item }}

} @empty {

No items available! Click the button to load items.

}

SWITCH Block

@switch (color) { @case ("Red") {
Selected color is Red
} @case ("Blue") {
Selected color is Blue
} @default {
Unknown color selected
} }

DEFER Block

The content below is deferred and will load on viewport visibility:

@defer (on viewport) {

Deferred Content

{{ heavyOperation() }}

} @placeholder {

Loading deferred content...

} @error {

Error loading deferred content!

}
